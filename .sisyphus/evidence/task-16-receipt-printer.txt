public function getDefaultPrinter(?Store $store = null): ?PrinterConfig
{
    $storeId = $store?->id;

    return PrinterConfig::forStore($storeId)
        ->default()
        ->active()
        ->first();
}

public function getAvailablePrinters(?Store $store = null): \Illuminate\Database\Eloquent\Collection
{
    $storeId = $store?->id;

    return PrinterConfig::forStore($storeId)
        ->active()
        ->get();
}
